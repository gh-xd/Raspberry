4.5 发送端py文件

4.5 发送端py文件

树莓派使用官方的CSI摄像头、利用picamera包，进行视频采集。 采集后的画面进行编码，然后发给视频接收端。

import cv2
import zmq
import base64
import picamera
from picamera.array import PiRGBArray

# 视频接收端的IP地址
IP = '192.168.0.16'

# 初始化摄像头
camera = picamera.PiCamera()
camera.resolution = (640, 480) # 分辨率
camera.framerate = 20 # 帧率
rawCapture = PiRGBArray(camera, size=(640, 480)) # ？

# 实例化一个用来发送帧的zmq对象
# 此处使用了tcp通信协议，其中5555为端口号，可自定义
# 要确保发送端和接收端的端口号一样
context = zmq.Context()
footage_socket = context.socket(zmq.PAIR)
footage_socket.connect('tcp://%s:5555'%IP)
print(IP)

# 循环从摄像头采集图像
# 因为使用的是树莓派摄像头，所以use_video_port=True
# 由于imencode()函数需要传入numpy array或者scalar来编码图像，所以，将采集到的frame转化为numpy array
for frame in camera.capture_continuous(rawCapture, format="bgr", use_video_port=True):
    frame_image = frame.array
    encoded, buffer = cv2.imencode('.jpg', frame_image) # 将frame编码成流数据，并保存到内存缓存
    jpg_as_text = base64.b64encode(buffer) # 缓存中的流数据通过base64编码后
    footage_socket.send(jpg_as_text) # 发送给视频接收端
    rawCapture.truncate(0) # 最后，释放资源为处理下一帧做准备
